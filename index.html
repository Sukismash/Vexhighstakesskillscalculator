    <script>

        function calculateScore() {

            const redRings = parseInt(document.getElementById("redRings").value);

            const blueRings = parseInt(document.getElementById("blueRings").value);

            const goalsInCorners = parseInt(document.getElementById("goalsInCorners").value);

            const hangLevel = parseInt(document.getElementById("hangLevel").value);


            let totalScore = 0;


            // Red rings scored: 1 point each

            totalScore += redRings * 1;


            // Blue rings scored (only count if red rings are scored under them): 3 points each

            if (redRings >= blueRings) {

                totalScore += blueRings * 3;

            }


            // Goals in corners: 5 points each

            totalScore += goalsInCorners * 5;


            // Hang level points

            if (hangLevel === 1) {

                totalScore += 3;

            } else if (hangLevel === 2) {

                totalScore += 6;

            } else if (hangLevel === 3) {

                totalScore += 12;

            }


            document.getElementById("result").innerText = "Total Score: " + totalScore;

        }

    </script>
